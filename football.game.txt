  <style>
    body {
      margin: 0;
      display: flex;
      justify-content: center;
      align-items: center;
      height: 100vh;
      background: green;
      font-family: Arial, sans-serif;
    }
    canvas {
      background: #87ceeb;
      border: 3px solid #fff;
    }
    #message {
      position: absolute;
      top: 20px;
      font-size: 20px;
      color: white;
      text-shadow: 1px 1px black;
    }
  </style>

<body>
  <div id="message">Press SPACE to shoot!</div>
  <canvas id="gameCanvas" width="500" height="400"></canvas>

  <script>
    const canvas = document.getElementById("gameCanvas");
    const ctx = canvas.getContext("2d");

    let ball = { x: 250, y: 350, radius: 12, vy: 0, shot: false };
    let goalKeeper = { x: 200, y: 50, width: 100, height: 20, vx: 3 };
    let score = 0;
    let gameOver = false;

    document.addEventListener("keydown", e => {
      if (e.code === "Space" && !ball.shot && !gameOver) {
        ball.vy = -6;
        ball.shot = true;
      }
      if (e.code === "Enter" && gameOver) restartGame();
    });

    function drawGoal() {
      ctx.fillStyle = "white";
      ctx.fillRect(100, 30, 300, 20);
    }

    function drawBall() {
      ctx.beginPath();
      ctx.arc(ball.x, ball.y, ball.radius, 0, Math.PI * 2);
      ctx.fillStyle = "white";
      ctx.fill();
      ctx.stroke();
    }

    function drawKeeper() {
      ctx.fillStyle = "red";
      ctx.fillRect(goalKeeper.x, goalKeeper.y, goalKeeper.width, goalKeeper.height);
    }

    function updateKeeper() {
      goalKeeper.x += goalKeeper.vx;
      if (goalKeeper.x <= 100 || goalKeeper.x + goalKeeper.width >= 400) {
        goalKeeper.vx *= -1;
      }
    }

    function updateBall() {
      if (ball.shot) {
        ball.y += ball.vy;

        if (
          ball.y - ball.radius <= goalKeeper.y + goalKeeper.height &&
          ball.x > goalKeeper.x &&
          ball.x < goalKeeper.x + goalKeeper.width
        ) {
          gameOver = true;
          document.getElementById("message").textContent = `Saved by the keeper! Final Score: ${score}. Press ENTER to try again.`;
        }

        if (ball.y <= 30) {
          score++;
          gameOver = true;
          document.getElementById("message").textContent = `GOAL! ðŸŽ‰ Score: ${score}. Press ENTER for next shot.`;
        }
      }
    }

    function drawScore() {
      ctx.fillStyle = "black";
      ctx.font = "20px Arial";
      ctx.fillText("Score: " + score, 10, 30);
    }

    function update() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      drawGoal();
      drawKeeper();
      drawBall();
      drawScore();

      if (!gameOver) {
        updateKeeper();
        updateBall();
      }
    }

    function restartGame() {
      ball = { x: 250, y: 350, radius: 12, vy: 0, shot: false };
      gameOver = false;
      document.getElementById("message").textContent = "Press SPACE to shoot!";
    }

    setInterval(update, 20);
  </script>
</body>
